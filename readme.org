** usage of ./judge.d

there are three supported test formats. here I introduce the easiest one.

- input file ./atcoder/abc081/b.d

#+begin_src d
/++TEST++
>>> Q1
3

this is wrong case
>>> Q2
4
5 6 8 10
===

<<<

>>> Q3
6
382253568 723152896 37802240 379425024 404894720 471526144
===
8
<<<
++TEST++/

import std.stdio;
import std.string;
import std.conv;
import std.algorithm;
import std.array;

int count2(int x) {
    int ret = 0;
    while (x % 2 == 0) {
        ++ret;
        x /= 2;
    }
    return ret;
}

void main() {
    size_t n;
    string s;
    readf("%d\n%s\n", &n, &s);
    auto c = s.split().map!(a => a.to!int.count2).reduce!min;
    writeln(c);
}
#+end_src
